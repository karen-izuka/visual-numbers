A Different Way to Write SQL Queries


      with a as (select date_key, sum(units_sold) units_sold
      from fact_table
      group by date_key),

      b as (select date_key, fiscal_year
      from date_table
      group by date_key, fiscal_year)
      
      select b.fiscal_year, sum(a.units_sold) units_sold
      from a
      left outer join b on (a.date_key = b.date_key)
      where b.fiscal_year = 2020
      group by b.fiscal_year;
